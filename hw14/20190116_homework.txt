   参考接口:
   
    struct RssItem
    {
        string title;
        string link;
        string description;
        string content;
    };
    
    class RssReader
    {
    public:
        RssReader();
        
        void parseRss();//解析

    private:
        vector<RssItem> _rss;
    };
